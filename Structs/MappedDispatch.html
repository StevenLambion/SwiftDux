MappedDispatch

@propertyWrapper
public struct MappedDispatch : DynamicProperty

Injects a function as a property in a view to dispatch actions to the provided store.

struct MyView : View {

  @MappedDispatch() var dispatch

  func handleClick() {
    dispatch(AppAction.doSomething())
  }

}
